# 适配器模式

```c#
namespace Adapter
{

    public enum AdapterType
    {
        None,
        Android,
        IOS
    }
    public class AndroidLine
    {
        public void Charge()
        {
            Debug.Log("借助安卓数据线充电...");
        }
    }
    public class IOSLine
    {
        public void Charge()
        {
            Debug.Log("借助IOS数据线充电...");
        }
    }
    public interface IAdapter
    {
        void Charge(AdapterType type);
    }

    public class Adapter : IAdapter
    {
        AndroidLine androidLine = new AndroidLine();
        IOSLine iosLine = new IOSLine();
        public void Charge(AdapterType type)
        {
            switch (type)
            {
                case AdapterType.None:
                    break;
                case AdapterType.Android:
                    androidLine.Charge();
                    break;
                case AdapterType.IOS:
                    iosLine.Charge();
                    break;
                default:
                    break;
            }
            
        }
    }
}
```
